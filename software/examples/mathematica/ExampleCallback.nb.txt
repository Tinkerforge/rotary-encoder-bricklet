Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="amb"(* Change to your UID *)

(* Create IPConnection and device object *)
ipcon=NETNew["Tinkerforge.IPConnection"]
re=NETNew["Tinkerforge.BrickletRotaryEncoder",uid,ipcon]
ipcon@Connect[host,port]

(* Set Period for count callback to 0.05s (50ms) *)
(* Note: The count callback is only called every 50ms if the *)
(*       count has changed since the last call! *)
re@SetCountCallbackPeriod[1000]

(* Callback function for count callback *)
CountCB[sender_,count_]:=
 Print["Count: "<>ToString[count]]
AddEventHandler[re@Count,CountCB]

Input["Click OK to exit"]

(* Clean up *)
ipcon@Disconnect[]
ReleaseNETObject[re]
ReleaseNETObject[ipcon]
